{
	SetSelector
	{
		techLogic none/Aurora Borealis
		set hotspot
		{
			version
			{
				Visual Special N Valkyrie_BirthA
				Connect ATTACHPOINT Aurora hotspot

			}
		}

			version

				Visual Special N Valkyrie_BirthA
